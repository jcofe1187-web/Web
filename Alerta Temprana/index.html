Flujo correcto de actualización de datos:
Los datos de estudiantes y docentes están embebidos en el archivo index.html. Cuando haya cambios en las listas, usted debe:
1. Venir acá → Actualizar listas / docentes → cargar los xlsx
2. Descargar el nuevo index.html generado
3. Subir ese archivo a GitHub (reemplazar el anterior)
Así todos los docentes ven los datos actualizados automáticamente.
function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var payload = JSON.parse(e.postData.contents); for (var i = 0; i < payload.reportes.length; i++) { var d = payload.reportes[i]; sheet.appendRow([ d.fecha, d.hora, d.nivel, d.seccion, d.cedula, d.estudiante, d.asignatura, d.docente, d.tiposAlerta, d.acciones, d.situacion, new Date().toLocaleString('es-CR') ]); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'activo'})) .setMimeType(ContentService.MimeType.JSON); }
